WITH ranked AS (
    SELECT 
        d.name AS Department, 
        e.name AS Employee,
        e.salary AS Salary,
        dense_rank() OVER (PARTITION BY departmentId ORDER BY salary DESC) AS dept_rank

    FROM employee e
    JOIN Department d ON e.departmentId = d.id
)

SELECT Department, Employee, Salary
FROM ranked
WHERE dept_rank <= 3;
